#pragma mark - Initialization

/**
 * Set's the access token and refresh token directly
 */
+ (void)initWithAccessToken:(NSString *)accessToken
            andRefreshToken:(NSString *)refreshToken
        withSuccessCallback:(void (^)(BOOL isSuccessful))successCallback
           andErrorCallback:(void (^)(SBError *error))errorCallback;

#pragma mark - Navigation and Presentation

/// Present the capture view controller over the current view controller.
+ (void)presentCaptureInCurrentViewController:(UIViewController *)controller;

/// Present the capture view controller over the current view controller using existing upload transaction.
+ (void)presentCaptureInCurrentViewController:(UIViewController *)controller usingTransaction:(SBLTransaction *)transaction;

/// Present the capture view controller over the current view controller with a default expense type.
+ (void)presentCaptureInCurrentViewController:(UIViewController *)controller withExpenseType:(SBLReceiptExpenseType)expenseType;

/// Present the capture view controller over the current view controller using existing upload transaction with a default expense type.
+ (void)presentCaptureInCurrentViewController:(UIViewController *)controller
                             usingTransaction:(SBLTransaction *)transaction
                              withExpenseType:(SBLReceiptExpenseType)expenseType;

/// Displays the Receipt View in the Current Navigation Controller.
+ (void)displayReceiptWithId:(NSString *)receiptId inNavigationController:(UINavigationController *)navigationController;

/// Displays the Export List In the Current Navigations Stack.
+ (void)pushToExportsControllerFromNavigationController:(UINavigationController *)navController;

/// Go To Search Receipt Controller.
+ (void)pushToSearchReceiptControllerFromController:(UIViewController *)controller andAllowFiltering:(BOOL)allowFilters;

#pragma mark - Error Handling

/// Receive CallBack Errors.
+ (void)registerErrorReceivingCallback:(void (^)(SBError *))callback;

#pragma mark - Analytics

/// Register analytics event through a listener.
+ (void)registerAnalyticsTrackerListener:(SBEventsListener *)listener;

#pragma mark - Reachability

/// Check if server is reachable for the configured environment.
+ (void)checkIfServerReachableForConfigurationWithCompletionHandler:(void (^)(bool isServerReachable))completionHandler;

#pragma mark - Localization

/// Attempts to provide localized string for given key. If SBSDKConfiguration.externalBundle is defined then search would be made first in strings provided in external bundle otherwise search would be made in SDK provided localized strings table.
+ (NSString *)localizedStringForKey:(NSString *)key;

@end
